12static uint32_t Platform_RCC_GetSystemClockFreq(void)
13{
14 uint32_t frequency = 0U;
15 LL_PLL_ClocksTypeDef PLL_Clocks;
16
17 /* Get SYSCLK source -------------------------------------------------------*/
18 switch (LL_RCC_GetSysClkSource())
19 {
20 /* No check on Ready: Won't be selected by hardware if not */
21 case LL_RCC_SYS_CLKSOURCE_STATUS_HSI:
22 frequency = HSI_VALUE >> (LL_RCC_HSI_GetDivider()>> RCC_CR_HSIDIV_Pos);
23 break;
24
25 case LL_RCC_SYS_CLKSOURCE_STATUS_CSI:
26 frequency = CSI_VALUE;
27 break;
28
29 case LL_RCC_SYS_CLKSOURCE_STATUS_HSE:
30 frequency = HSE_VALUE;
31 break;
32
33 case LL_RCC_SYS_CLKSOURCE_STATUS_PLL1:
34 LL_RCC_GetPLL1ClockFreq(&PLL_Clocks);
35 frequency = PLL_Clocks.PLL_P_Frequency;
36 break;
37
38 default:
39 /* Nothing to do */
40 break;
41 }
42
43 return frequency;
44}
